Step 5: Database Table Structure Examples
Tables WITH Tracking Support:

CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    status TINYINT(1) DEFAULT 1,
    updated_at DATETIME,
    updated_by INT
);

CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT,
    status TINYINT(1) DEFAULT 1,
    updated_at DATETIME,
    updated_by INT,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

Tables WITHOUT Tracking Support:

CREATE TABLE categories (
    id INT PRIMARY KEY AUTO_INCREMENT,
    status TINYINT(1) DEFAULT 1
    -- No updated_at or updated_by fields
);

CREATE TABLE products (
    id INT PRIMARY KEY AUTO_INCREMENT,
    category_id INT,
    status TINYINT(1) DEFAULT 1,
    FOREIGN KEY (category_id) REFERENCES categories(id)
    -- No updated_at or updated_by fields
);
